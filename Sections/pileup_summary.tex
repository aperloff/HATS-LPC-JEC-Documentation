\subsection{Pileup Summary}
%---------------------------------------------------------------------------------------------------------------------------------------
\begin{frame}[t]\frametitle{Pileup Summary}
    \vspace*{-0.35cm}
    \begin{block}{}
    	\vbox to .99\vsize{
    	\begin{itemize}
    		\item In-time and out-of-time pileup are \textbf{additional pp interactions} leading to unwanted calorimeter deposits and tracks
    		\item We simulate pileup and apply a \textbf{reweighting} procedure to match the pileup distribution in data
    		\item Various \textbf{pileup mitigation techniques} exist to remove the effects of pileup
    		\item \textbf{PUPPI} is a pileup mitigation technique that accounts for local variations and neutral particles
    	\end{itemize}
    	}
    	\vspace*{-1.4cm}
    \end{block}


\end{frame}
